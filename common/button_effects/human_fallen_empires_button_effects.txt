# Scope type varies depending on what is selected
# This = selected object or player country
# From = player country

human_fallen_empires_button_effect_leaders_recrutment_block = {
	potential = {
		exists = from
		from = {
			or = {
				human_fallen_empires_trigger_leaders_recrutment_enabled = yes
				has_country_flag = human_fallen_empires_leaders_hire_block
			}
		}
	}
	allow = {
		custom_tooltip = {
			text = human_fallen_empires_leaders_button_disable
			always = no
		}
	}
	effect = {
	}
}
