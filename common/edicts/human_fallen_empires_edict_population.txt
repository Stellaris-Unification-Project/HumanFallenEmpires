# country edicts
country_edict = {
	name = human_fallen_empires_population_allow_for_caste
	cost = { influence = 1 }
	potential = {
		not = {
			has_global_flag = human_fallen_empires_population_disable
		}
	}
	allow = {
		any_owned_pop = {
			has_citizenship_type = {
				country = root
				type = citizenship_caste_system
			}
			has_migration_control = no
		}
	}

	ai_weight = {
		weight = 0
	}
}

# evacuate
planet_edict = {
	name = human_fallen_empires_population_evacuate
	cost = { influence = 100 }
	length = 0
	potential = {
		nor = {
			has_global_flag = human_fallen_empires_population_hide_edicts
			has_global_flag = human_fallen_empires_population_disable
		}
		owner = {
			any_owned_planet = {
				not = {
					is_planet = root
				}
				free_pop_tiles > 0
			}
		}
	}
	allow = { }
	effect = {
		every_owned_pop = {
			limit = {
				exists = owner
				is_growing = no
			}
			save_event_target_as = human_fallen_empires_settle_pop_target
			owner = {
				human_fallen_empires_effect_settle_pop = yes
			}
		}
	}

	ai_weight = {
		weight = 0
	}
}

# hide all
planet_edict = {
	name = human_fallen_empires_population_hide_planet_edicts
	cost = { influence = 0 }
	length = 0
	potential = {
		not = {
			has_global_flag = human_fallen_empires_population_disable
		}
	}

	effect = {
		if = {
			limit = {
				has_global_flag = human_fallen_empires_population_hide_edicts
			}
			remove_global_flag  = human_fallen_empires_population_hide_edicts
			else = {
				set_global_flag = human_fallen_empires_population_hide_edicts
			}
		}
	}
	ai_weight = {
		weight = 0
	}
}
