ap_engineered_evolution = {
	on_enabled = {
		add_research_option = tech_gene_seed_purification
		hidden_effect = {
			country_event = { id = utopia.2700 }
		}
	}
	possible = {
		custom_tooltip = {
			fail_text = "synthetic_empire_biological_ascension"
			NOT = { has_trait = trait_mechanical }
		}
		custom_tooltip = {
			fail_text = "requires_ascension_perks_1"
			num_ascension_perks > 0
		}
		custom_tooltip = {
			fail_text = "requires_technology_gene_tailoring"
			has_technology = tech_gene_tailoring
		}


		custom_tooltip = {
			fail_text = human_fallen_empires_ascension_points
			not = {
				calc_true_if = {
					amount = 2
					has_ascension_perk = ap_engineered_evolution
					has_ascension_perk = ap_evolutionary_mastery
					has_ascension_perk = ap_the_flesh_is_weak
					has_ascension_perk = ap_synthetic_evolution
					has_ascension_perk = ap_mind_over_matter
					has_ascension_perk = ap_transcendence
				}
			}
		}
	}
	modifier = {
		trait_points = 2
		modify_species_cost_mult = -0.50
	}

	potential = {
		NOT = {
			has_ascension_perk = ap_engineered_evolution
		}
	}

	ai_weight = {
		factor = 50
		modifier = {
			factor = 10
			or = {
				has_country_flag = human_fallen_empires_egypt
				has_country_flag = human_fallen_empires_future
			}
		}
	}
}
