# egypt
# dont like varelse
human_fallen_empires_opinion_egypt_vs_varelse = {
	trigger = {
		or = {
			and = {
				has_country_flag = human_fallen_empires_egypt
				from = {
					is_same_species = event_target:human_fallen_empires_varelse_species
				}
			}
			and = {
				 from = {
					 has_country_flag = human_fallen_empires_egypt
				 }
				 is_same_species = event_target:human_fallen_empires_varelse_species
			}
		}
	}
	opinion = {
		base = -100
	}
}

# future

		has_country_flag = human_fallen_empires_future

		from = {

			}
		}
	}
	opinion = {
		base = -100
	}
}

human_fallen_empires_opinion_future_vs_bad = {

		}

		base = -30
		modifier = {

			}
		}
	}
}
