human_fallen_empires_effect_refuge_or_die = {
	#log = "pop refuge effect"
	if = {
		limit = {
			is_robot_pop = no
			nor = {
				has_trait = trait_hive_mind
				has_trait = trait_nerve_stapled
			}
			exists = owner
			owner = {
				any_relation = {
					num_pops > 0
					human_fallen_empires_trigger_is_awake = yes
					any_owned_planet = {
						human_fallen_empires_trigger_planet_valid_for_settle = yes
					}
				}
			}
		}
		save_event_target_as = human_fallen_empires_refuge_pop_target
		owner = {
			random_relation = {
				limit = {
					num_pops > 0
					human_fallen_empires_trigger_is_awake = yes
					any_owned_planet = {
						human_fallen_empires_trigger_planet_valid_for_settle = yes
						habitability = {
							who = event_target:human_fallen_empires_refuge_pop_target
							value >= 0.2
						}
					}
				}
				weights = {
					base = 10
					modifier = {
						mult = 0
						has_policy_flag = refugees_not_allowed
						event_target:human_fallen_empires_refuge_pop_target = {
							not = {
								is_same_species = prev
							}
						}
					}
					modifier = {
						mult = 0
						has_policy_flag = refugees_only_citizens
						event_target:human_fallen_empires_refuge_pop_target = {
							nor = {
								is_same_species = prev
								has_citizenship_type = {
									type = citizenship_full
									country = prev
								}
								has_citizenship_type = {
									type = citizenship_caste_system
									country = prev
								}
								has_citizenship_type = {
									type = citizenship_organic_trophy
									country = prev
								}
							}
						}
					}
					modifier = {
						mult = 100
						has_country_flag = human_fallen_empires_refuge_country_prior@prev
					}
					modifier = {
						mult = 8
						event_target:human_fallen_empires_refuge_pop_target = {
							or = {
								has_citizenship_type = {
									type = citizenship_full
									country = prev
								}
								has_citizenship_type = {
									type = citizenship_caste_system
									country = prev
								}
								has_citizenship_type = {
									type = citizenship_organic_trophy
									country = prev
								}
							}
						}
					}
					modifier = {
						mult = 2
						event_target:human_fallen_empires_refuge_pop_target = {
							or = {
								has_citizenship_type = {
									type = citizenship_limited
									country = prev
								}
								has_citizenship_type = {
									type = citizenship_caste_system_limited
									country = prev
								}
							}
						}
					}
					modifier = {
						mult = 2
						has_policy_flag = refugees_allowed
					}
					modifier = {
						mult = 2
						any_owned_planet = {
							human_fallen_empires_trigger_planet_valid_for_settle = yes
							habitability = {
								who = event_target:human_fallen_empires_refuge_pop_target
								value >= 0.6
							}
						}
					}
					modifier = {
						mult = 2
						any_owned_planet = {
							human_fallen_empires_trigger_planet_valid_for_settle = yes
							habitability = {
								who = event_target:human_fallen_empires_refuge_pop_target
								value >= 0.4
							}
						}
					}
					modifier = {
						mult = 0.5
						event_target:human_fallen_empires_refuge_pop_target = {
							has_citizenship_type = {
								type = citizenship_assimilation
								country = prev
							}
						}
					}
					modifier = {
						mult = 0.25
						event_target:human_fallen_empires_refuge_pop_target = {
							has_citizenship_type = {
								type = citizenship_slavery
								country = prev
							}
						}
					}
				}
				event_target:human_fallen_empires_refuge_pop_target = {
					remove_modifier = pop_recently_conquered
					add_modifier = {
						modifier = recent_refugee
						days = 3600
					}
					save_event_target_as = human_fallen_empires_settle_pop_target
				}
				human_fallen_empires_effect_settle_pop = yes
			}
		}
	}
	else = {
		log = "refuge pop killed"
		kill_pop = yes
	}
}

# for country need human_fallen_empires_settle_pop_target
human_fallen_empires_effect_settle_pop = {
	if = {
		limit = {
			not = {
				any_owned_planet = {
					human_fallen_empires_trigger_planet_valid_for_settle = yes
					# habitability = {
						# who = event_target:human_fallen_empires_settle_pop_target
						# value >= 0.2
					# }
				}
			}
		}
		event_target:human_fallen_empires_settle_pop_target = {
			log = "looping in settle pop"
			human_fallen_empires_effect_refuge_or_die = yes
		}
	}
	else = {
		random_owned_planet = {
			limit = {
				human_fallen_empires_trigger_planet_valid_for_settle = yes
				# habitability = {
					# who = event_target:human_fallen_empires_settle_pop_target
					# value > 0.2
				# }
			}
			weights = {
				base = 10
				modifier = {
					mult = 2
					habitability = {
						who = event_target:human_fallen_empires_settle_pop_target
						value > 0.8
					}
				}
				modifier = {
					mult = 2
					habitability = {
						who = event_target:human_fallen_empires_settle_pop_target
						value > 0.7
					}
				}
				modifier = {
					mult = 2
					habitability = {
						who = event_target:human_fallen_empires_settle_pop_target
						value > 0.6
					}
				}
				modifier = {
					mult = 2
					habitability = {
						who = event_target:human_fallen_empires_settle_pop_target
						value > 0.5
					}
				}
				modifier = {
					mult = 2
					habitability = {
						who = event_target:human_fallen_empires_settle_pop_target
						value > 0.4
					}
				}
				modifier = {
					mult = 2
					habitability = {
						who = event_target:human_fallen_empires_settle_pop_target
						value > 0.3
					}
				}
			}
			event_target:human_fallen_empires_settle_pop_target = {
				purge = no
			}
			resettle_pop = {
				pop = event_target:human_fallen_empires_settle_pop_target
				planet = this
			}
			add_organic_sanctuaries = yes
			#log = "pop moved to [this.GetName]"
		}
	}
}

human_fallen_empires_effect_add_hyperlanes = {
	while = {
		count = 3
		random_system = {
			limit = {
				nor = {
					has_hyperlane_to = prev
					is_same_value = prev
				}
				distance = {
					source = prev
					type = euclidean
					min_distance >= 0
					max_distance <= 20
					use_bypasses = no
				}
			}
			add_hyperlane = {
				from = prev
				to = this
			}
		}
	}
	if = {
		limit = {
			not = {
				any_system = {
					has_hyperlane_to = prev
				}
			}
		}
		random_system = {
			limit = {
				not = {
					is_same_value = prev
				}
				distance = {
					source = prev
					type = euclidean
					min_distance >= 0
					max_distance <= 50
					use_bypasses = no
				}
			}
			add_hyperlane = {
				from = prev
				to = this
			}
		}
	}
}

					civics = {

	}
	else_if = {
		limit = {
			host_has_dlc = "Utopia"

				ethic = ethic_gestalt_consciousness
			}
			authority = auth_hive_mind
		}

			30 = {
				change_government = {
					civics = random
				}
			}

					civics = {
						civic = civic_hive_devouring_swarm

					}
				}

				is_spiritualist = yes
			}
			prev = {
				set_subject_of = {
					who = prev
					subject_type = dominion
				}
			}
		}
		else_if = {
			limit = {
				is_materialist = yes
			}
			prev = {
				set_subject_of = {
					who = prev
					subject_type = satellite
				}
			}
		}
	}
	else_if = {
		limit = {
			exists = event_target:human_fallen_empires_coalition_leader
			is_country = event_target:human_fallen_empires_coalition_leader
		}
		prev = {
			set_subject_of = {
				who = prev
				subject_type = human_fallen_empires_future_vassal
			}
			human_fallen_empires_effect_advisor_future_on = yes
		}
	}
	else_if = {
		limit = {
			exists = event_target:human_fallen_empires_egypt_country
			is_country = event_target:human_fallen_empires_egypt_country
		}
		prev = {
			set_subject_of = {
				who = prev
				subject_type = human_fallen_empires_egypt_vassal
			}
		}
	}
	else = {
		prev = {
			set_subject_of = {
				who = prev
				subject_type = vassal
			}
		}
	}
}
