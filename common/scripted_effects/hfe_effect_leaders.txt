# for country
hfe_leaders_calc_governor_capacity = {
	# log = "calc governor capacity"
	if = {
		limit = {
			num_pops = 0
		}
		set_variable = {
			which = hfe_leaders_governor_capacity
			value = 0
		}
	}
	else = {
		set_variable = {
			which = hfe_leaders_governor_capacity
			value = 1
		}
		every_sector = {
			root = {
				change_variable = {
					which = hfe_leaders_governor_capacity
					value = 1.1
				}
			}
		}
		if = {
			limit = {
				has_ethic = ethic_pacifist
			}
			change_variable = {
				which = hfe_leaders_governor_capacity
				value = 1
			}
		}
		else_if = {
			limit = {
				has_ethic = ethic_fanatic_pacifist
			}
			change_variable = {
				which = hfe_leaders_governor_capacity
				value = 2
			}
		}
		if = {
			limit = {
				has_tradition = tr_domination_colonial_viceroys
			}
			change_variable = {
				which = hfe_leaders_governor_capacity
				value = 2
			}
		}
		if = {
			limit = {
				has_tradition = tr_harmony_finish
			}
			change_variable = {
				which = hfe_leaders_governor_capacity
				value = 0.5
			}
		}
		if = {
			limit = {
				has_tradition = tr_expansion_finish
			}
			change_variable = {
				which = hfe_leaders_governor_capacity
				value = 0.5
			}
		}
		if = {
			limit = {
				has_tradition = tr_prosperity_finish
			}
			change_variable = {
				which = hfe_leaders_governor_capacity
				value = 0.5
			}
		}
		# log = "[this.hfe_leaders_governor_capacity]"
		every_owned_leader = {
			limit = {
				leader_class = governor
			}
			root = {
				change_variable = {
					which = hfe_leaders_governor_capacity
					value = -1
				}
			}
		}
		# log = "[this.hfe_leaders_governor_capacity]"
	}
}
