human_fallen_empires_effect_brainworm_set_free_pop_with_reservation = {
	if = {
		limit = {
			not = {
				has_building = human_fallen_empires_brainworm_reservation_0
			}
		}
		if = {
			limit = {
				any_tile = {
					has_pop = no
					has_deposit = no
				}
			}
			random_tile = {
				limit = {
					has_pop = no
					has_deposit = no
				}
				remove_blocker = yes
				set_building = human_fallen_empires_brainworm_reservation_0
				create_pop = {
					species = owner
				}
				pop = {
					modify_species = {
						species = this
						remove_trait = human_fallen_empires_species_trait_brainworm
					}
				}
			}
			else = {
				random_tile = {
					limit = {
						has_pop = no
					}
					remove_blocker = yes
					clear_deposits = yes
					set_building = human_fallen_empires_brainworm_reservation_0
					create_pop = {
						species = owner
					}
					pop = {
						modify_species = {
							species = this
							remove_trait = human_fallen_empires_species_trait_brainworm
						}
					}
				}
			}
		}
	}
}
