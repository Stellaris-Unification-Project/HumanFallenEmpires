# Triggers event when a system is entered by ships from a country (not pirates or monsters) for the first time
# Scope = Ship
# From = System
# FromFrom = Country
on_entering_system_first_time = {
	events = {
		human_fallen_empires.600
	}
	random_events = {
		49 = 0
		# varelse fleet event
		1 = human_fallen_empires.51
	}
}

# Triggered when two empires discover each other
on_first_contact = {
	events = {
		human_fallen_empires.551
		human_fallen_empires.553
		# contact with egypt
		human_fallen_empires.4
		# contact with future
		human_fallen_empires.5000
	}
}

# This = owner of ship 1 (destroyed)
# From = owner of ship 2 (combatant)
# FromFrom = ship 1
# FromFromFrom = ship 2

	events = {
		# discovery killed end game
		human_fallen_empires.552
		# remove modifier then dron outpost destroyed
		human_fallen_empires.602
		# kill pirate base
		human_fallen_empires.653
	}
}
