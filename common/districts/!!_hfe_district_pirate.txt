@base_cost = 300

hfe_district_pirate_alloys = {
	base_buildtime = 240
    is_capped_by_modifier = no
	show_on_uncolonized = {
		is_planet_class = hfe_pc_pirate_base
	}
	potential = {
		is_planet_class = hfe_pc_pirate_base
	}
	resources = {
		category = planet_districts
		cost = {
			minerals = @base_cost
		}
	}
	planet_modifier = {
        planet_housing_add = 5
        job_foundry_add = 3
        job_artisan_add = 2
	}
	ai_weight = {
		weight = 1
		modifier = {
			factor = 3
			owner = {
				has_monthly_income = {
					resource = alloys
					value < 0
				}
			}
		}
		modifier = {
			factor = 3
			owner = {
				has_monthly_income = {
					resource = consumer_goods
					value < 0
				}
			}
		}
		modifier = {
			factor = 2
			free_housing < 0
		}
		modifier = {
			factor = 3
			any_owned_pop = {
				OR = {
					is_unemployed = yes
					has_job = servant
				}
			}
		}
		# No one to work it anyway
		modifier = {
			factor = 0
			free_housing > 0
			NOT = {
				any_owned_pop = {
					OR = {
						is_unemployed = yes
						has_job = servant
					}
				}
			}
		}
	}
	ai_resource_production = {
        alloys = 1
        consumer_goods = 1
		trigger = {
			always = yes
		}
	}
}
