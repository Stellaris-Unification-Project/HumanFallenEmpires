human_fallen_empires_agenda_discovery_science_faction = {
	weight_modifier = {
		weight = 10000
		modifier = {
			factor = 0
			not = {
				leader_class = scientist
			}
		}
		modifier = {
			factor = 0
			not = {
				from = {
					is_country_type = human_fallen_empires_discovery
				}
			}
		}
	}

	modifier = {
		country_resource_physics_research_add = 15
		country_resource_society_research_add = 15
		country_resource_engineering_research_add = 15
	}
}
