human_fallen_empires_cb_total_war = {
	show_in_diplomacy = yes
	show_notification = yes
	potential = {
		num_owned_planets > 0
		num_pops > 0
		or = {
			human_fallen_empires_trigger_is_fallen = yes
			is_country_type = default
		}
		or = {
			human_fallen_empires_trigger_is_awake = yes
			is_ai = no
		}
	}
	is_valid = {
		nand = {
			exists = overlord
			overlord = {
				is_country = from
			}
		}
		from = {
			num_pops > 0
			or = {
				human_fallen_empires_trigger_is_fallen = yes
				and = {
					is_country_type = default
					prev = {
						has_technology = human_fallen_empires_tech_total_war
					}
					relative_power = {
						who = prev
						category = fleet
						value > equivalent
					}
				}
			}
		}
	}
}
