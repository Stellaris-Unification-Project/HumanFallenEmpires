		set_subject_of = {
			who = none
		}
		every_relation = {
			limit = {
				or = {
					is_subject_type = human_fallen_empires_future_vassal
					is_country = from
				}
			}
			set_subject_of = {
				who = root
				subject_type = human_fallen_empires_future_vassal
			}
		}
		save_global_event_target_as = human_fallen_empires_coalition_leader
	}

		weight = 1000
		modifier = {
			factor = 0
			relative_power = {
				who = from
				category = fleet
				value < equivalent
			}

	}
}
