hfe_map_mode_plague = {
	icon = "GFX_map_mode_default"
	enable_terra_incognita = no
	can_change_point_of_view = no
	#shortcut = "CTRL+z"

	color = {
		value = hsv { 0.0 0.0 0.8 0.7 }
		custom_tooltip = "MAP_MODE_NO_CONTACT"
		condition = {
			NOT = { has_established_contact = FROM }
			#always = no
		}
	}
	# haven't plague
	color = {
		value = hsv { 0.25 1 0.6 1.0 }
		custom_tooltip = hfe_map_mode_plague_not_infected
		condition = {
			not = {
				any_owned_planet = {
					has_planet_flag = hfe_plague_planet
				}
			}
		}
	}
	# have plague
	color = {
		value = hsv { 0.0 0.9 0.6 1.0 }
		custom_tooltip = hfe_map_mode_plague_infected
		condition = {
			any_owned_planet = {
				has_planet_flag = hfe_plague_planet
			}
		}
	}
}
