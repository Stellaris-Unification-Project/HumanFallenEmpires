# Works in species scope, with a country in from
is_lonely_hive_mind_pop = {
	or = {
		and = {
			has_trait = trait_hive_mind
			NOT = { from = { has_authority = auth_hive_mind } }
			from = {
				NOR = {
					is_country_type = fallen_empire
					is_country_type = awakened_fallen_empire
				}
			}
		}
		and = {
			human_fallen_empires_trigger_is_brainworm_species = yes
			from = {
				has_policy_flag = human_fallen_empires_policies_brainworm_0.2
			}
		}
	}
}
