		has_country_flag = human_fallen_empires_discovery.151.4
		#has_country_flag = human_fallen_empires_discovery.151.crew_menu
		has_country_flag = human_fallen_empires_discovery.151.contract_menu
	}
}

# for country
human_fallen_empires_trigger_discovery_docking_any_quest = {
	or = {
		has_country_flag = human_fallen_empires_discovery_contract_embassy_success
		has_country_flag = human_fallen_empires_discovery_contract_embassy_fail
	}
}

human_fallen_empires_trigger_discovery_docking_t0_cost = {
	custom_tooltip = {
		text = human_fallen_empires_discovery.151.have_enough_resources
		or = {
			and = {
				event_target:human_fallen_empires_discovery_docking_planet.owner = {
					not = {
						is_country = root
					}
				}
				resource_stockpile_compare = { resource = minerals value >= @human_fallen_empires_discovery_docking_update_minerals_cost_t0 }
				resource_stockpile_compare = { resource = energy value >= @human_fallen_empires_discovery_docking_update_energy_cost_t0 }
			}
			and = {
				event_target:human_fallen_empires_discovery_docking_planet.owner = {
					is_country = root
				}
				resource_stockpile_compare = { resource = minerals value >= @human_fallen_empires_discovery_docking_update_minerals_cost_t0_own }
				resource_stockpile_compare = { resource = energy value >= @human_fallen_empires_discovery_docking_update_energy_cost_t0_own }
			}
		}
	}
}
