# annex occupying systems planet
# A planets controller becomes a country not the same as the owner.
# Root = Planet
# From = Planet Owner
# FromFrom = Planet Controller (the one occupying)
planet_event = {
	id = hfe_total_war.51
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		exists = owner
		exists = controller
		exists = solar_system
		nor = {
			is_controlled_by = owner
			solar_system = {
				any_system_planet = {
					exists = owner
					owner = {
						not = {
							is_country_type = primitive
						}
					}
					is_controlled_by = owner
				}
			}
		}
		owner = {
			any_war = {
				any_war_participant = {
					is_country = root.owner
					is_war_leader = no
				}
				any_war_participant = {
					is_country = root.controller
				}
				# war leader
				any_war_participant = {
					prev = {
						using_war_goal = {
							type = hfe_wg_total_war
							owner = prev
						}
					}
					or = {
						is_in_federation_with = root.controller
						is_country = root.controller
						any_subject = {
							is_country = root.controller
						}
					}
				}
			}
		}
	}

	immediate = {
		conquer = controller
		if = {
			limit = {
				exists = starbase
			}
			starbase = {
				set_owner = root.controller
			}
		}
	}
}

	id = hfe_total_war.1010
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		host_has_dlc = "Leviathans Story Pack"
		not = {
			has_global_flag = hfe_war_in_heaven_started
		}
		count_country = {
			count >= 2
			limit = {
				is_country_type = awakened_fallen_empire
				or = {
					is_at_war = no
					has_global_flag = hfe_war_in_heaven_started_by_ai
				}
				fleet_power > 40000
			}
		}

	immediate = {
		log = "war in heavens started"
