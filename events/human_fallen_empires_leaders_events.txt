namespace = human_fallen_empires_leaders

# calc leaders capacity

country_event = {
	id = human_fallen_empires_leaders.0
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		not = {
			has_global_flag = human_fallen_empires_disable_leaders
		}
		or = {
			is_ai = no
			is_country_type = default
		}
	}

	immediate = {
		if = {
			limit = {
				not = {
					has_modifier = human_fallen_empires_leaders_started
				}
			}
			add_modifier = {
				modifier = human_fallen_empires_leaders_started
				days = -1
			}
		}
		human_fallen_empires_leaders_calc_leaders_num = yes
		human_fallen_empires_leaders_calc_leaders_capacity = yes
		if = {
			limit = {
				not = {
					any_owned_leader = {
						leader_class = governor
					}
				}
			}
			country_event = {
				id = human_fallen_empires_leaders.50
			}
		}
		if = {
			limit = {
				not = {
					any_owned_leader = {
						leader_class = scientist
					}
				}
			}
			country_event = {
				id = human_fallen_empires_leaders.51
			}
		}
		if = {
			limit = {
				not = {
					any_owned_leader = {
						leader_class = admiral
					}
				}
			}
			country_event = {
				id = human_fallen_empires_leaders.52
			}
		}
		if = {
			limit = {
				not = {
					any_owned_leader = {
						leader_class = general
					}
				}
			}
			country_event = {
				id = human_fallen_empires_leaders.53
			}
		}
	}
}

# gain new trait from mod by on_leader_level_up
country_event = {
	id = human_fallen_empires_leaders.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		from = {
			if = {
				limit = {
					species = {
						has_trait = trait_psionic
					}
				}
				human_fallen_empires_effect_add_psionic_trait = yes
			}
			if = {
				limit = {
					num_traits < 5
				}
				random_list = {
					90 = {
					}
					10 = {
						modifier = {
							factor = 1.5
							has_level = 3
						}
						modifier = {
							factor = 2.5
							has_level = 4
						}
						modifier = {
							factor = 5
							has_level > 4
						}
						human_fallen_empires_leaders_add_random_trait = yes
					}
				}
			}
		}
	}
}

# gain governor 50
country_event = {
	id = human_fallen_empires_leaders.50
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		check_variable = {
			which = human_fallen_empires_leaders_governor_capacity
			value > 0
		}
	}

	immediate = {
		random_pool_leader = {
			limit = {
				leader_class = governor
			}
			exile_leader_as = human_fallen_empires_leaders_governor
		}
		create_fleet = {
			effect = {
				set_owner = root
				create_ship = {
					name = random
					random_existing_design = science
				}
				set_location = root.capital_scope
				set_leader = human_fallen_empires_leaders_governor
				leader = {
					unassign_leader = this
				}
				delete_fleet = this
			}
		}
		change_variable = {
			which = human_fallen_empires_leaders_governor_capacity
			value = -1
		}
		country_event = {
			id = human_fallen_empires_leaders.54
		}
	}
}
