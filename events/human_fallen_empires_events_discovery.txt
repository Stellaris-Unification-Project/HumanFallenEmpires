			species = this

			message = no
			tech = human_fallen_empires_tech_discovery_ship
		}
		capital_scope = {
			save_event_target_as = human_fallen_empires_discovery_start_location
			clear_buildings = yes
			destroy_colony = yes
			reroll_planet = yes
		}
		create_leader = {
			type = admiral
			species = this
			name = random
			skill = 0
		}
		create_ship_design = {
			design = Deepvoid
		}
		add_ship_design = last_created_design
		create_fleet = {
			name = Discovery
			effect = {
				set_owner = prev
				create_ship = {
					name = Discovery
					design = Deepvoid
					effect = {
						set_ship_flag = human_fallen_empires_discovery_ship
					}
				}
				assign_leader = last_created_leader
				set_location = event_target:human_fallen_empires_discovery_start_location
			}
		}
		every_owned_ship = {
			repair_ship = yes
			add_modifier = {
				modifier = human_fallen_empires_unkeep_80
				days = -1
			}
		}
		add_minerals = 500
		add_energy = 500
	}
}
